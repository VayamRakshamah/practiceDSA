package Solutions;

class Solution {
    public int removeDuplicates(int[] nums) {
        int left =0;
        int right = 1;
        int k = 0;
        while((left<right) && (right<nums.length)){
            if(nums[left]==nums[right]){
                right++;
            }else{
                ++left;
                nums[left]=nums[right];
                right++;
            }
        }
        return left+1;
    }
}
